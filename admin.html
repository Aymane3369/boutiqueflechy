document.getElementById('uploadArea').style.display = 'none'; document.getElementById('emojiSelector').style.display = 'block'; selectedImage = null; } }); // --- Gestion des radios --- const radioEmoji = modalDiv.querySelector('input[value="emoji"]'); product.id : Date.now().toString(), name, description, type: 'simple', price, stock, icon: icon, image_url: image_url, active }; if (isEdit) { const idx = products.findIndex(p => p.id === product.id); if (idx !== -1) products[idx] = newProd; } else products.push(newProd); } else { const basePrice = parseFloat(modalDiv.querySelector('#basePrice').value); if (isNaN(basePrice)) return alert('Prix de base requis'); const finalVariants = []; const rows = variantsContainer.querySelectorAll('.variant-row'); rows.forEach(row => { const inputs = row.querySelectorAll('input'); const attrsStr = inputs[0]?.value || ''; const [taille, couleur] = attrsStr.split('/'); const stock = parseInt(inputs[1]?.value || 0); const price = parseFloat(inputs[2]?.value || 0) || basePrice; if (taille.trim()) { finalVariants.push({ id: Date.now() + Math.random().toString(36).substring(2,6), attributes: { taille: taille.trim(), couleur: couleur.trim() || '' }, stock, price }); } }); if (finalVariants.length === 0) return alert('Ajoutez au moins une variante'); const newProd = { id: isEdit ?
